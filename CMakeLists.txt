cmake_minimum_required(VERSION 3.16)

project(AwesomeNbtViewer
    VERSION 0.1.0
    LANGUAGES C CXX
)

### Additional flags
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

### Version
set(AWESOMENBTVIEWER_VERSION_MAJOR ${PROJECT_VERSION_MAJOR})
set(AWESOMENBTVIEWER_VERSION_MINOR ${PROJECT_VERSION_MINOR})
set(AWESOMENBTVIEWER_VERSION_PATCH ${PROJECT_VERSION_PATCH})
set(AWESOMENBTVIEWER_VERSION ${AWESOMENBTVIEWER_VERSION_MAJOR}.${AWESOMENBTVIEWER_VERSION_MINOR}.${AWESOMENBTVIEWER_VERSION_PATCH})

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

### Installation directories
set(AwesomeNbtViewer_INSTALL_BINARY_DIR "bin")
set(AwesomeNbtViewer_INSTALL_INCLUDE_DIR "include")
set(AwesomeNbtViewer_INSTALL_LIBRARY_DIR "lib/${CMAKE_BUILD_TYPE}")

find_package(Qt6 REQUIRED COMPONENTS Widgets)

find_package(AwesomeMC REQUIRED)

### AwesomeNbtViewer
add_subdirectory(src)
